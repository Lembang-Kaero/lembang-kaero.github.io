document.addEventListener("DOMContentLoaded", function () { const headerElement = document.querySelector('header'); if (headerElement) { // Karena file .js ini berada di folder 'assets/components', // kita langsung panggil 'header.html' tanpa menuliskan ulang nama foldernya. const antiCacheToken = new Date().getTime(); const headerUrl = 'header.html?v=' + antiCacheToken; fetch(headerUrl) .then(response => { if (!response.ok) throw new Error('Gagal memuat komponen header'); return response.text(); }) .then(htmlContent => { headerElement.innerHTML = htmlContent; // MENYALAKAN WARNA KUNING PADA MENU BERITA / AKTIF try { const currentPath = window.location.pathname.toLowerCase(); const navLinks = headerElement.querySelectorAll('.navbar-nav .nav-link'); navLinks.forEach(link => { const linkHref = link.getAttribute('href').toLowerCase(); if (currentPath.includes(linkHref) || (currentPath.endsWith('/') && linkHref === 'index.html')) { link.classList.add('active', 'text-warning'); } else { link.classList.remove('active', 'text-warning'); } }); } catch (navError) { console.warn("Deteksi menu aktif dilewati:", navError); } }) .catch(error => console.error('Kendala sistem Header:', error)); } });
